import "UnityEngine"

function main()

	local button = GameObject.Find("Button")
	local buttonscript = button:GetComponent(Button)
	buttonscript.onClick:AddListener(

			print("button")
		end
	)

	--local imageScript = image:GetComponent(Image)

end
